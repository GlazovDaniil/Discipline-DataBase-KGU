SELECT  Направление, ФИО,  L.ЗначВидКонтр, N.ЗначВидКонтр FROM
	(SELECT DISTINCT Направление, ФИО,  S.ЗначВидКонтр, КодЭкзаменатора, КодНапр FROM
		(SELECT DISTINCT P.КодЭкзаменатора, N.ИндНомКаф, Фамилия + ' ' + Имя + ' ' + Отчество ФИО, Направление, КодНапр
		FROM Prepodavatel P JOIN NaprPodgotovki N
		ON P.ИндНомКаф = N.ИндНомКаф) F
		CROSS JOIN 
	(SELECT ЗначВидКонтр FROM VidKontrola) S) L
LEFT JOIN
	(SELECT КодНапр, КодЭкзаменатора, ЗначВидКонтр FROM [Group] G JOIN
		(SELECT КодГр, КодЭкзаменатора, ЗначВидКонтр FROM Vedomost Ved JOIN VidKontrola Vid
							ON Vid.КодВидКонтр = Ved.КодВидКонтр) V
		ON G.КодГр = V.КодГр) N
ON L.КодЭкзаменатора = N.КодЭкзаменатора AND L.ЗначВидКонтр = N.ЗначВидКонтр AND L.КодНапр = N.КодНапр


//Как раюотает:
В первой половине (перед левым соединением) собирается таблица с названиями направлений и их кодов, ФИО преподавателей и их коды
а так же полученный результат перемножается с видами контроля

Во второй половине (после левого соединения) формируется таблица с кодами преподавателей, которые принимали контроли, а так же 
коды направлений, студенты которых проходили контроль 

В итоге запрос формирует таблицу путем левого соединения двух ранее полученных таблиц с условием что коды направлений, коды экзаменатора
и значения вида контроля совпадали
